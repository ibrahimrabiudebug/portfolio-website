Weather Data Management System

class WeatherAPI {
    constructor() {
        this.baseURL = 'https://api.openweathermap.org/data/2.5';
        this.apiKey = 'YOUR_API_KEY'; // Securely managed
        this.cache = new Map();
        this.cacheDuration = 15 * 60 * 1000; // 15 minutes
    }

    async fetchWeatherData(city, units = 'metric') {
        const cacheKey = `${city}_${units}`;
        const cached = this.cache.get(cacheKey);
        
        if (cached && Date.now() - cached.timestamp < this.cacheDuration) {
            return cached.data;
        }

        try {
            const [current, forecast] = await Promise.all([
                this.fetchCurrentWeather(city, units),
                this.fetchForecast(city, units)
            ]);

            const data = { current, forecast };
            this.cache.set(cacheKey, {
                timestamp: Date.now(),
                data
            });

            return data;
        } catch (error) {
            console.error('API Error:', error);
            throw new Error('Failed to fetch weather data');
        }
    }

    async fetchCurrentWeather(city, units) {
        const response = await fetch(
            `${this.baseURL}/weather?q=${city}&units=${units}&appid=${this.apiKey}`
        );
        
        if (!response.ok) throw new Error('City not found');
        return await response.json();
    }

    async fetchForecast(city, units) {
        const response = await fetch(
            `${this.baseURL}/forecast?q=${city}&units=${units}&appid=${this.apiKey}`
        );
        
        if (!response.ok) throw new Error('Forecast unavailable');
        return await response.json();
    }

    processWeatherData(rawData) {
        return {
            temperature: Math.round(rawData.main.temp),
            feelsLike: Math.round(rawData.main.feels_like),
            humidity: rawData.main.humidity,
            pressure: rawData.main.pressure,
            windSpeed: Math.round(rawData.wind.speed * 3.6), // Convert m/s to km/h
            windDirection: this.getWindDirection(rawData.wind.deg),
            visibility: (rawData.visibility / 1000).toFixed(1),
            description: rawData.weather[0].description,
            icon: rawData.weather[0].icon,
            sunrise: new Date(rawData.sys.sunrise * 1000),
            sunset: new Date(rawData.sys.sunset * 1000)
        };
    }

    getWindDirection(degrees) {
        const directions = ['N', 'NE', 'E', 'SE', 'S', 'SW', 'W', 'NW'];
        const index = Math.round(degrees / 45) % 8;
        return directions[index];
    }
}
