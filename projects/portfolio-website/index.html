Interactive Skill Visualization

/**
 * Animate skill bars on scroll with Intersection Observer
 * Demonstrates modern JavaScript patterns and performance optimization
 */
const animateSkillBars = () => {
    const skillBars = document.querySelectorAll('.skill-progress');
    
    const observer = new IntersectionObserver((entries) => {
        entries.forEach(entry => {
            if (entry.isIntersecting) {
                const skillBar = entry.target;
                const targetWidth = skillBar.getAttribute('data-width');
                
                // Animate progress bar
                skillBar.style.width = `${targetWidth}%`;
                skillBar.style.transition = 'width 1.5s cubic-bezier(0.4, 0, 0.2, 1)';
                
                // Unobserve after animation completes
                observer.unobserve(skillBar);
            }
        });
    }, {
        threshold: 0.3,
        rootMargin: '0px 0px -50px 0px'
    });
    
    skillBars.forEach(bar => observer.observe(bar));
};

// Initialize on DOM content loaded
document.addEventListener('DOMContentLoaded', animateSkillBars);
